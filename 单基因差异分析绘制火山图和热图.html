sorted_df  <- as.data.frame(filtered_deg_limma) %>%

4.3 比如我想找到上调前十名的基因，并列出来，执行如下操作。首先确定矩阵中上调第一名的基因名是什么，然后定位到这个基因，确定其所在行和列，然后列出后面10个基因名称

sorted_matrix_dec <- sorted_matrix[order(sorted_matrix[, 2], decreasing = TRUE), ] #重复排序操作，可忽略
target <- "VWF" #确定第一个基因为VWF
target_col <- 7 # 定义第一个基因的列数
positions <- which(sorted_matrix_dec[, target_col] == target, arr.ind = TRUE)
if (length(positions) > 0) {
  cat("在第", target_col, "列中找到目标字符 '", target, "' 的位置：\n", sep = "")
  print(data.frame(行号 = positions, 列号 = target_col))
} else {
  cat("在第", target_col, "列中未找到目标字符 '", target, "'.\n", sep = "")
} #找到目标基因行数和列数

6 pheatmap绘制热图

6.1 首先是准备热图需要的数据，其实就是表达矩阵的可视化而已。

deg_genes <- deg_limma %>%
 filter(abs(logFC)>1, adj.P.Val < 0.05) %>% 
 pull(genesymbol)

heat_df <- expr[deg_genes,]

# 自己进行标准化的好处就是可以自定义范围
heat_df <- t(scale(t(heat_df)))
heat_df[heat_df >= 3] <- 3
heat_df[heat_df < -3] <- -3

# 列注释
anno_col <- data.frame(sample_group = sample_group)
rownames(anno_col) <- colnames(heat_df)
